Options -MultiViews
<IfModule mod_rewrite.c>
  RewriteEngine On
  # Deixar assets passarem sem rewrite
  RewriteRule ^assets/ - [L]
  # SPA routing: só redireciona se não for arquivo/pasta real
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule ^ /index.html [L]
</IfModule>
