Options -MultiViews
<IfModule mod_rewrite.c>
  RewriteEngine On
  # Assets passam sem rewrite
  RewriteRule ^assets/ - [L]
  # SPA: redireciona rotas inexistentes para index.html
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule ^ /index.html [L]
</IfModule>
# Fallback para LiteSpeed que ignora RewriteRule em alguns casos
ErrorDocument 404 /index.html
